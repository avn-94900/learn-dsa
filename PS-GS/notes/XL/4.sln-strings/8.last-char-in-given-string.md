##  Find Last Index of a Character in a String

public class LastIndexFinder {
    public static int findLastIndex(String input, char target) {
        return input.lastIndexOf(target);
    }

    public static void main(String[] args) {
        String input = "hello world";
        char target = 'o';
        System.out.println("Last index of '" + target + "': " + findLastIndex(input, target));
    }
}

public class LastIndexFinder {
    public static int findLastIndex(String input, char target) {
        // Start from the end and move backwards
        for (int i = input.length() - 1; i >= 0; i--) {
            if (input.charAt(i) == target) {
                return i; // return immediately when found
            }
        }
        return -1; // if not found
    }

    public static void main(String[] args) {
        String input = "hello world";
        char target = 'o';
        System.out.println("Last index of '" + target + "': " + findLastIndex(input, target));
    }
}
