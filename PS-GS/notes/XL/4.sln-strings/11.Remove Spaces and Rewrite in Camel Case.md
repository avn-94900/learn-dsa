* Split by spaces.

* Time: **O(n)**

public class CamelCaseConverter {
    public static String toCamelCase(String input) {
        String[] words = input.trim().split("\\s+"); // split by one or more spaces
        if (words.length == 0) return "";

        StringBuilder sb = new StringBuilder(words[0].toLowerCase()); // first word in lowercase

        for (int i = 1; i < words.length; i++) {
            // no need to check for " " since split("\\s+") removes them
            sb.append(Character.toUpperCase(words[i].charAt(0)))
              .append(words[i].substring(1).toLowerCase());
        }
        return sb.toString();
    }

    public static void main(String[] args) {
        String input = "   hello   world   program   ";
        System.out.println("CamelCase: " + toCamelCase(input));
    }
}
