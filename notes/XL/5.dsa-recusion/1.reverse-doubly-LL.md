

###  Full Code: Doubly Linked List with Iterative Reversal

```java
class DoublyLinkedList {
    // Node class
    static class Node {
        int data;
        Node prev, next;

        Node(int data) {
            this.data = data;
            this.prev = null;
            this.next = null;
        }
    }

    Node head;

    // Insert at end
    public void append(int data) {
        Node newNode = new Node(data);
        if (head == null) {
            head = newNode;
            return;
        }
        Node temp = head;
        while (temp.next != null) {
            temp = temp.next;
        }
        temp.next = newNode;
        newNode.prev = temp;
    }

    // Print DLL forward
    public void printList(Node node) {
        Node temp = node;
        while (temp != null) {
            System.out.print(temp.data + " ");
            temp = temp.next;
        }
        System.out.println();
    }

    // Reverse DLL iteratively (your method)
    public void reverseIterative() {
        Node prevPointer = null;
        Node currPointer = head;
        Node nextPointer = null;

        while (currPointer != null) {
            nextPointer = currPointer.next;   // store next

            currPointer.next = prevPointer;   // reverse next
            currPointer.prev = nextPointer;   // fix prev

            prevPointer = currPointer;        // move prev ahead
            currPointer = nextPointer;        // move curr ahead
        }

        head = prevPointer; // new head
    }

    // Driver
    public static void main(String[] args) {
        DoublyLinkedList dll = new DoublyLinkedList();

        // ✅ Test Case 1
        dll.append(1);
        dll.append(2);
        dll.append(3);
        dll.append(4);
        dll.append(5);

        System.out.println("Original DLL (Test 1):");
        dll.printList(dll.head);

        dll.reverseIterative();

        System.out.println("Reversed DLL (Test 1):");
        dll.printList(dll.head);

        // ✅ Test Case 2
        DoublyLinkedList dll2 = new DoublyLinkedList();
        dll2.append(10);
        dll2.append(20);
        dll2.append(30);

        System.out.println("\nOriginal DLL (Test 2):");
        dll2.printList(dll2.head);

        dll2.reverseIterative();

        System.out.println("Reversed DLL (Test 2):");
        dll2.printList(dll2.head);
    }
}
```
